```
WITH RankedSales AS (
    SELECT
        CustomerId,
        SaleDate,
        SaleAmount,
        ROW_NUMBER() OVER(
            PARTITION BY CustomerId
            ORDER BY SaleAmount DESC, SaleDate DESC
        ) AS rn
    FROM Sales
    WHERE SaleDate >= DATEADD(year, -1, GETDATE())
),
CustomerMaxSaleDetails AS (
    SELECT
        CustomerId,
        SaleDate,
        SaleAmount
    FROM RankedSales
    WHERE rn = 1
)
SELECT
    cmsd.CustomerId,
    cmsd.SaleDate,
    cmsd.SaleAmount    AS MaxSaleAmount,
    DENSE_RANK() OVER (
        ORDER BY cmsd.SaleAmount DESC
    )                  AS SalesRank
FROM CustomerMaxSaleDetails cmsd
ORDER BY
    SalesRank ASC,
    cmsd.SaleAmount DESC;
```
